Customizer configuration

To replace our design with your desired theme. Please do configuration as mention

Path : src > app > shared > services > layout > layout.services.ts 

  
     public config = {
        settings: {
           layout_type: '{{layoutService.config.settings.layout_type}}',
           layout_version: '{{layoutService.config.settings.layout_version}}',
           sidebar_type: '{{layoutService.config.settings.sidebar_type}}',
           icon: '{{layoutService.config.settings.icon}}',
        },
        color: {
           primary_color: '{{layoutService.config.color.primary_color}}',
           secondary_color: '{{layoutService.config.color.secondary_color}}'
        }
     }
